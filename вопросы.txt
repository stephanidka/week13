let today = Date.now();
let tomorrow = new Date(2023, 6, 25, 0, 0);
let milliseconds = tomorrow - today;

function howManySeconds() {
  let seconds = Math.floor(milliseconds / 1000); // Конвертируем миллисекунды в секунды и округляем вниз
  return seconds;
}

let seconds = howManySeconds();
console.log(seconds);
